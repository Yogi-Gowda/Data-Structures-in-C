#include <stdio.h>
#include <stdlib.h>

struct Node {
    int data;
    struct Node* next;
};

struct Node* insertAtFront(struct Node* head, int value) {
    struct Node* newNode = (struct Node*)malloc(sizeof(struct Node));
    newNode->data = value;
    newNode->next = head;
    return newNode;
}

void displayList(struct Node* head) {
    struct Node* temp = head;
    while (temp != NULL) {
        printf("%d -> ", temp->data);
        temp = temp->next;
    }
    printf("NULL\n");
}

    struct Node* head = NULL;

    head = insertAtFront(head, 20);
    head = insertAtFront(head, 10);
    
    printf("List after inserting at front: ");
    displayList(head);
    
    return 0;
}

#include <stdio.h>
#include <stdlib.h>

struct Node {
    int data;
    struct Node* next;
};

struct Node* insertAtFront(struct Node* head, int value) {
    struct Node* newNode = (struct Node*)malloc(sizeof(struct Node));
    newNode->data = value;
    newNode->next = head;
    return newNode;
}

struct Node* insertAtPosition(struct Node* head, int value, int position) {
    if (position < 1) {
        printf("Error: Position must be positive!\n");
        return head;
    }
    
    if (position == 1) {
        return insertAtFront(head, value);
    }
    
    struct Node* newNode = (struct Node*)malloc(sizeof(struct Node));
    newNode->data = value;
    
    struct Node* temp = head;
    int count = 1;
    
    while (temp != NULL && count < position - 1) {
        temp = temp->next;
        count++;
    }
    
    if (temp == NULL) {
        printf("Error: Position %d is out of range!\n", position);
        free(newNode);
        return head;
    }
    
    newNode->next = temp->next;
    temp->next = newNode;
    return head;
}

void displayList(struct Node* head) {
    struct Node* temp = head;
    while (temp != NULL) {
        printf("%d -> ", temp->data);
        temp = temp->next;
    }
    printf("NULL\n");
}

int main() {
    struct Node* head = NULL;
    
    head = insertAtFront(head, 30);
    head = insertAtFront(head, 10);
    
    printf("Original list: ");
    displayList(head);
    
    head = insertAtPosition(head, 20, 2);
    printf("After inserting 20 at position 2: ");
    displayList(head);
    
    head = insertAtPosition(head, 5, 1);
    printf("After inserting 5 at position 1: ");
    displayList(head);
    
    // Test error case
    head = insertAtPosition(head, 100, 10);
    
    return 0;
}
